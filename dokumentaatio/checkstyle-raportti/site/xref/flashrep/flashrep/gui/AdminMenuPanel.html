1   package flashrep.flashrep.gui;
2   
3   import flashrep.flashrep.logic.AppControlLogic;
4   import java.awt.Component;
5   import static java.awt.Component.CENTER_ALIGNMENT;
6   import java.awt.GridLayout;
7   import javax.swing.BoxLayout;
8   import javax.swing.JButton;
9   import javax.swing.JLabel;
10  import javax.swing.JList;
11  import javax.swing.JPanel;
12  import javax.swing.JScrollPane;
13  import javax.swing.ListSelectionModel;
14  
15  /**
16   *
17   * Adminvalikkoluokka josta hallinnoidaan ohjelman käyttäjiä.
18   */
19  public class AdminMenuPanel extends JPanel {
20  
21      private Views views;
22      private AppControlLogic controller;
23      private JList userlist;
24      private UsersModel model;
25  
26      /**
27       * Luokan kontrolleri joka saa käyttöönsä parametreina annettavat
28       * kontrollerin ja näkymiä hallinnoivan Viewsin.
29       *
30       * @param views Ikkunanäkymiä hallinnoiva views-luokan ilmentymä
31       * @param controller Kontrolleri
32       */
33      public AdminMenuPanel(Views views, AppControlLogic controller) {
34          this.views = views;
35          this.controller = controller;
36          initComponents();
37      }
38  
39      //Luodaan valikon komponentit
40      private void initComponents() {
41          // luodaan komponentit
42  
43          BoxLayout mainLayout = new BoxLayout(this, BoxLayout.Y_AXIS);
44          this.setLayout(mainLayout);
45  
46          //Aloitusruudun teksti luodaan, keskitetään ja lisätään
47          JLabel headLineLabel = new JLabel("Tervetuloa pääkäyttäjä!");
48          headLineLabel.setAlignmentX(Component.CENTER_ALIGNMENT);
49          add(headLineLabel);
50          add(new JLabel("\n"));
51  
52          JLabel userCountLabel = new JLabel("Ohjelmalla on " + this.controller.getUsers().getUsercount() + " käyttäjää.");
53          userCountLabel.setAlignmentX(Component.CENTER_ALIGNMENT);
54          add(userCountLabel);
55          add(new JLabel("\n"));
56  
57          //Luodaan ja lisätään paneeleja jotta saadaan jotenkin haluttu näkö
58          JPanel adminMenuPanel = new JPanel(new GridLayout(0, 2, 10, 10));
59          JPanel listPanel = new JPanel();
60          JPanel buttonsPanelWrapper = new JPanel();
61          BoxLayout listPanelLayout = new BoxLayout(listPanel, BoxLayout.Y_AXIS);
62          BoxLayout buttonsPanelWrapperLayout = new BoxLayout(buttonsPanelWrapper, BoxLayout.Y_AXIS);
63          listPanel.setLayout(listPanelLayout);
64          buttonsPanelWrapper.setLayout(buttonsPanelWrapperLayout);
65          JPanel buttonsPanel = new JPanel(new GridLayout(8, 0, 20, 20));
66  
67          adminMenuPanel.add(listPanel);
68          adminMenuPanel.add(buttonsPanelWrapper);
69          buttonsPanelWrapper.add(new JLabel("\n"));
70          buttonsPanelWrapper.add(buttonsPanel);
71          add(adminMenuPanel);
72  
73          this.userlist = new JList();
74          this.model = this.controller.getUserList();
75          this.userlist.setModel(model);
76  
77          this.userlist.setSelectionMode(ListSelectionModel.SINGLE_INTERVAL_SELECTION);
78          this.userlist.setLayoutOrientation(JList.VERTICAL);
79          this.userlist.setVisibleRowCount(10);
80          this.userlist.setFixedCellHeight(20);
81          this.userlist.setFixedCellWidth(140);
82          JScrollPane listScroller = new JScrollPane(userlist);
83  
84          // Luodaan tekstit ja painikkeet sekä lisätään ne
85          JLabel userlistLabel = new JLabel("Käyttäjät");
86          JButton getPasswordButton = new JButton("Näytä käyttäjän salasana");
87          JButton setPasswordButton = new JButton("Aseta käyttäjän salasana");
88          JButton removeUserButton = new JButton("Poista käyttäjä(t)");
89          JButton signOutButton = new JButton("Kirjaudu ulos");
90  
91          userlistLabel.setAlignmentX(CENTER_ALIGNMENT);
92          listPanel.add(userlistLabel);
93          listPanel.add(listScroller);
94  
95          buttonsPanel.add(getPasswordButton);
96          buttonsPanel.add(setPasswordButton);
97          buttonsPanel.add(removeUserButton);
98          buttonsPanel.add(signOutButton);
99  
100         //Lisätään komentotunnisteet kuuntelija varten
101         getPasswordButton.setActionCommand("getPass");
102         setPasswordButton.setActionCommand("setPass");
103         removeUserButton.setActionCommand("remove");
104         signOutButton.setActionCommand("signOut");
105 
106         //Luodaan kuuntelija ja lisätään se nappuloiden ja listan käyttöön
107         AdminMenuListener listener = new AdminMenuListener(views, controller, userCountLabel, userlist, model, getPasswordButton, setPasswordButton, removeUserButton, signOutButton);
108         userlist.addListSelectionListener(listener);
109         getPasswordButton.addActionListener(listener);
110         setPasswordButton.addActionListener(listener);
111         removeUserButton.addActionListener(listener);
112         signOutButton.addActionListener(listener);
113     }
114 }
