1   package flashrep.flashrep.gui;
2   
3   import flashrep.flashrep.logic.AppControlLogic;
4   import java.awt.Container;
5   import java.awt.Dimension;
6   import java.awt.Toolkit;
7   import javax.swing.BorderFactory;
8   import javax.swing.JFrame;
9   import javax.swing.JOptionPane;
10  import javax.swing.WindowConstants;
11  
12  /**
13   * Käyttöliittymäluokka
14   */
15  public class GUI implements Runnable {
16  
17      private JFrame frame;
18      private AppControlLogic controller;
19  
20      /**
21       * Luokan konstruktori joka injektoi käyttöliittymän käyttöön parametrina
22       * annettavan kontrollerin.
23       *
24       * @param controller Kontrolleri käyttäjä- ja korttiluokkien käyttöä varten
25       */
26      public GUI(AppControlLogic controller) {
27          this.controller = controller;
28      }
29  
30      /**
31       * Metodi käynnistää käyttöliittymän.
32       */
33      @Override
34      public void run() {
35          frame = new JFrame("FlashRep");
36          frame.setPreferredSize(new Dimension(800, 400));
37  
38          frame.setDefaultCloseOperation(WindowConstants.DO_NOTHING_ON_CLOSE);
39  
40          initComponents(frame.getContentPane());
41  
42          frame.pack();
43          frame.setVisible(true);
44  
45          centerFrame();
46  
47          frame.addWindowListener(new java.awt.event.WindowAdapter() {
48              @Override
49              public void windowClosing(java.awt.event.WindowEvent windowEvent) {
50                  Object[] buttons = {"Kyllä", "Ei"};
51                  if (JOptionPane.showOptionDialog(frame,
52                          "Haluatko tosiaan sulkea ohjelman?", "Sulje ohjelma?",
53                          JOptionPane.YES_NO_OPTION,
54                          JOptionPane.QUESTION_MESSAGE, null, buttons, buttons[1]) == JOptionPane.YES_OPTION) {
55                      controller.getDataHandler().saveData();
56                      System.exit(0);
57                  }
58              }
59          });
60      }
61  
62      private void initComponents(Container container) {
63          // lisätään näkymät
64          Views views = new Views(controller);
65          views.setBorder(BorderFactory.createEmptyBorder(15, 15, 15, 15));
66          container.add(views);
67      }
68  
69      /* Tämä suoraan Jarmo Kallion ohjelmasta SInkTheShips, 
70      keskittää aloittaessa ohjelman ruudun keskelle.
71       */
72      private void centerFrame() {
73          Dimension xy = Toolkit.getDefaultToolkit().getScreenSize();
74          this.frame.setLocation(xy.width / 2 - this.frame.getWidth() / 2, xy.height / 2 - this.frame.getHeight() / 2);
75      }
76  
77      /**
78       * Metodi palauttaa käyttöliittymän ikkunan.
79       *
80       * @return
81       */
82      public JFrame getFrame() {
83          return frame;
84      }
85  
86  }
