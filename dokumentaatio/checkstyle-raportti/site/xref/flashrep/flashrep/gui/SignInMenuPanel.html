19  /**
20   * JPanelin toiminnot perivä luokka joka toteuttaa kirjautumisvalikon.
21   */
22  public class SignInMenuPanel extends JPanel {
23  
24      private Views views;
25      private Controller controller;
26  
27      /**
28       * Luokan konstruktori joka injektoi kontrollerin.
29       *
30       * @param views Ikkunanäkymät
31       * @param controller  Kontrolleri käyttäjä- ja korttiluokkien käyttöä varten
32       */
33      public SignInMenuPanel(Views views, Controller controller) {
34          BoxLayout layout = new BoxLayout(this, BoxLayout.Y_AXIS);
35          this.setLayout(layout);
36          this.views=views;
37          this.controller=controller;
38          initComponents();
39      }
40  
41      //Luodaan valikon komponentit
42      private void initComponents() {
43  
44          //Aloitusruudun teksti luodaan, keskitetään ja lisätään
45          JLabel label1 = new JLabel("Kirjaudu tai luo tunnus");
46          label1.setAlignmentX(Component.CENTER_ALIGNMENT);
47          add(label1);
48        
49          //Lisätään paneeli paneelin sisälle jotta saadaan haluttu tyyli
50          JPanel signInMenuPanel = new JPanel(new GridLayout(0, 2,10,10));
51          add(signInMenuPanel);
52  
53          //Luodaan tekstit, kentät ja painikkeet
54          JLabel usernameText = new JLabel("Käyttäjätunnus");
55          JTextField usernameField = new JTextField(20);
56  
57          JLabel areUaUserText = new JLabel("Onko sinulla jo käyttäjätunnus?");
58          JLabel empty = new JLabel("");
59  
60          //Jos on jo tunnukset
61          JRadioButton isUserButton = new JRadioButton("Kyllä, salasanani on:");
62  
63          isUserButton.setSelected(true);
64          JPasswordField isUserPasswordField = new JPasswordField(20);
65  
66          //Kun ei ole vielä tunnuksia
67          JRadioButton isNotUserButton = new JRadioButton("Ei, haluan luoda tunnukset!");
68  
69          //Lisätään valintanapit ryhmään
70          ButtonGroup buttonGroup = new ButtonGroup();
71          buttonGroup.add(isUserButton);
72          buttonGroup.add(isNotUserButton);
73  
74          /*Uuden käyttäjän salasanan luonti on aluksi piilossa 
75          kunnes käyttäjä ilmoittaa, ettei hänellä ole tunnuksia
76           */
77          JLabel createPasswordText = new JLabel("Keksi salasana:");
78          createPasswordText.setVisible(false);
79          JPasswordField createPasswordField = new JPasswordField(20);
80          createPasswordField.setVisible(false);
81          JLabel repeatPasswordText = new JLabel("Toista salasana:");
82          repeatPasswordText.setVisible(false);
83          JPasswordField repeatPasswordField = new JPasswordField(20);
84          repeatPasswordField.setVisible(false);
85  
86          //Kirjautuminen/tunnusten luontipainike
87          JButton signInOrCreateUserButton = new JButton("Kirjaudu");
88  
89          //Komentotunnisteet kuuntelijaa varten 
90          isUserButton.setActionCommand("isUser");
91          isNotUserButton.setActionCommand("isNotUser");
92          signInOrCreateUserButton.setActionCommand("pushed");
93  
94          //lisätään komponentit paneeliin
95          signInMenuPanel.add(usernameText);
96          signInMenuPanel.add(usernameField);
97          signInMenuPanel.add(areUaUserText);
98          signInMenuPanel.add(empty);
99          signInMenuPanel.add(isUserButton);
100         signInMenuPanel.add(isUserPasswordField);
101         signInMenuPanel.add(isNotUserButton);
102         signInMenuPanel.add(new JLabel(""));
103         signInMenuPanel.add(createPasswordText);
104         signInMenuPanel.add(createPasswordField);
105         signInMenuPanel.add(repeatPasswordText);
106         signInMenuPanel.add(repeatPasswordField);
107         signInMenuPanel.add(new JLabel(""));
108         signInMenuPanel.add(new JLabel(""));
109         signInMenuPanel.add(new JLabel(""));
110         signInMenuPanel.add(signInOrCreateUserButton);
111 
112         //lisätään kuuntelija ja asetetaan se eri painikkeille      
113         SignInMenuListener listener = new SignInMenuListener(usernameField, isUserPasswordField, createPasswordText, createPasswordField, repeatPasswordText, repeatPasswordField, signInOrCreateUserButton, views, controller);
114         isUserButton.addActionListener(listener);
115         isNotUserButton.addActionListener(listener);
116         signInOrCreateUserButton.addActionListener(listener);
117     }
