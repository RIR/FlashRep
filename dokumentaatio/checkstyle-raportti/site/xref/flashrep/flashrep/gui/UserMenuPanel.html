18  /**
19   * Käyttäjävalikkoluokka.
20   */
21  public class UserMenuPanel extends JPanel {
22  
23      private Views views;
24      private AppControlLogic controller;
25      private JList collectionlist;
26      private CollectionsModel model;
27  
28      /**
29       * Luokan konstruktori jolle annetaan parametreina näkymiä hallitseva
30       * Views-luokan olio sekä kortti- ja käyttäjäluokkia hallinnoiva
31       * kontrolleri.
32       *
33       * @param views Views-luokan olio
34       * @param controller Kontrolleri
35       */
36      public UserMenuPanel(Views views, AppControlLogic controller) {
37          this.views = views;
38          this.controller = controller;
39          initComponents();
40      }
41  
42      //Luodaan valikon komponentit
43      private void initComponents() {
44          // luodaan komponentit
45  
46          BoxLayout mainLayout = new BoxLayout(this, BoxLayout.Y_AXIS);
47          this.setLayout(mainLayout);
48  
49          //Aloitusruudun teksti luodaan, keskitetään ja lisätään
50          JLabel label1 = new JLabel("Käyttäjä: " + this.controller.getCurrentUser().toString());
51          label1.setAlignmentX(Component.CENTER_ALIGNMENT);
52          add(label1);
53          for (int i = 0; i < 3; i++) {
54              add(new JLabel("\n"));
55          }
56  
57          //Luodaan ja lisätään kömpelösti paneeleja jotta saadaan jotenkin haluttu näkö
58          JPanel userMenuPanel = new JPanel(new GridLayout(0, 2, 10, 10));
59          JPanel listPanel = new JPanel();
60          JPanel buttonsPanelWrapper = new JPanel();
61          BoxLayout listPanelLayout = new BoxLayout(listPanel, BoxLayout.Y_AXIS);
62          BoxLayout buttonsPanelWrapperLayout = new BoxLayout(buttonsPanelWrapper, BoxLayout.Y_AXIS);
63          listPanel.setLayout(listPanelLayout);
64          buttonsPanelWrapper.setLayout(buttonsPanelWrapperLayout);
65          JPanel buttonsPanel = new JPanel(new GridLayout(8, 0, 20, 20));
66  
67          userMenuPanel.add(listPanel);
68          userMenuPanel.add(buttonsPanelWrapper);
69          buttonsPanelWrapper.add(new JLabel("\n"));
70          buttonsPanelWrapper.add(buttonsPanel);
71          add(userMenuPanel);
72  
73          this.collectionlist = new JList(); //data has type Object[]
74          this.model = this.controller.getCurrentUsersCollections();
75          this.collectionlist.setModel(model);
76  
77          this.collectionlist.setSelectionMode(ListSelectionModel.SINGLE_INTERVAL_SELECTION);
78          this.collectionlist.setLayoutOrientation(JList.VERTICAL);
79          this.collectionlist.setVisibleRowCount(10);
80          this.collectionlist.setFixedCellHeight(20);
81          this.collectionlist.setFixedCellWidth(140);
82          JScrollPane listScroller = new JScrollPane(collectionlist);
83  
84          // Luodaan tekstit ja painikkeet sekä lisätään ne
85          JLabel ownCollectionsLabel = new JLabel("Omat kokoelmat");
86          JButton studyNowButton = new JButton("Opiskele nyt");
87          JButton renameButton = new JButton("Nimeä uudelleen");
88          JButton removeCollectionButton = new JButton("Poista kokoelma");
89          JButton createNewCollectionButton = new JButton("Luo uusi kokoelma");
90          JButton signOutButton = new JButton("Kirjaudu ulos");
91  
92          ownCollectionsLabel.setAlignmentX(CENTER_ALIGNMENT);
93          listPanel.add(ownCollectionsLabel);
94          listPanel.add(listScroller);
95  
96          buttonsPanel.add(studyNowButton);
97          buttonsPanel.add(renameButton);
98          buttonsPanel.add(createNewCollectionButton);
99          buttonsPanel.add(removeCollectionButton);
100         buttonsPanel.add(signOutButton);
101 
102         //Lisätään komentotunnisteet kuuntelija varten
103         studyNowButton.setActionCommand("study");
104         renameButton.setActionCommand("rename");
105         removeCollectionButton.setActionCommand("remove");
106         createNewCollectionButton.setActionCommand("create");
107         signOutButton.setActionCommand("signOut");
108 
109         //Luodaan kuuntelija ja lisätään se nappuloiden ja listan käyttöön
110         UserMenuListener listener = new UserMenuListener(views, controller, collectionlist, model, studyNowButton, renameButton, createNewCollectionButton, removeCollectionButton, signOutButton);
111         collectionlist.addListSelectionListener(listener);
112         studyNowButton.addActionListener(listener);
113         renameButton.addActionListener(listener);
114         removeCollectionButton.addActionListener(listener);
115         createNewCollectionButton.addActionListener(listener);
116         signOutButton.addActionListener(listener);
117 
118     }
